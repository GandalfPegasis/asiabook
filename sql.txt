CREATE TABLE profile(
    id int primary key auto_increment,
    name text not null,
    email text not null,
    password text not null,
    birth_date date,
    nationality VARCHAR(100),
    role ENUM("student", "teacher"),
    department varchar(100),
    language text,
    contact_info text
);

CREATE TABLE friends(
    id int primary key auto_increment,
    profile_id_1 int,
    profile_id_2 int,
    FOREIGN KEY (profile_id_1) REFERENCES profile(id) ON UPDATE cascade ON DELETE cascade,
    FOREIGN KEY (profile_id_2) REFERENCES profile(id) ON UPDATE cascade ON DELETE cascade
);

CREATE TABLE friend_request(
    id int primary key auto_increment,
    requested_by int,
    requested_to int,
    FOREIGN KEY (requested_by) REFERENCES profile(id) ON UPDATE cascade ON DELETE cascade,
    FOREIGN KEY (requested_to) REFERENCES profile(id) ON UPDATE cascade ON DELETE cascade
);

CREATE TABLE posts(
    id int primary key auto_increment,
    posted_by int,
    caption text,
    likes int default 0,
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP;
    FOREIGN KEY (posted_by) REFERENCES profile(id) ON UPDATE cascade ON DELETE cascade
);

CREATE TABLE post_tags(
    id int primary key auto_increment,
    post_id int,
    tag_id
    FOREIGN KEY (post_id) REFERENCES posts(id) ON UPDATE cascade ON DELETE cascade
)

create table comment_post(
    id int primary key auto_increment,
    comment_by int,
    post_id int,
    content text NOT NULL,
    FOREIGN KEY (comment_by) REFERENCES profile(id) ON UPDATE cascade ON DELETE cascade,
    FOREIGN KEY (post_id) REFERENCES posts(id) ON UPDATE cascade ON DELETE cascade,
)

CREATE TABLE post_picture(
    id int primary key auto_increment,
    post_id int,
    location text,
    FOREIGN KEY (post_id) REFERENCES posts(id) ON UPDATE cascade ON DELETE cascade
);

CREATE TABLE forum(
    id int primary key auto_increment,
    title varchar(100),
    post_by int,
    description text,
    club_id int DEFAULT NULL,
    votes int NOT NULL DEFAULT 0,
    FOREIGN KEY (post_by) REFERENCES profile(id) ON UPDATE cascade ON DELETE cascade,
    FOREIGN KEY (club_id) REFERENCES clubs(id) ON UPDATE cascade ON DELETE cascade
);

    FOREIGN KEY (user_id) REFERENCES profile(id) ON UPDATE cascade ON DELETE cascade
);

CREATE TABLE forum_votes (
    id int NOT NULL,
    forum_id int NOT NULL,
    user_id int NOT NULL,
    vote tinyint NOT NULL
);

CREATE TABLE messages(
    id int primary key auto_increment,
    sender_id int NOT NULL,
    receiver_id int NOT NULL,
    content text NOT NULL,
    is_read boolean NOT NULL DEFAULT false,
    created_at datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (sender_id) REFERENCES profile(id) ON UPDATE cascade ON DELETE cascade,
    FOREIGN KEY (receiver_id) REFERENCES profile(id) ON UPDATE cascade ON DELETE cascade

SELECT 
    p.id AS post_id,
    p.caption,
    prof.name AS author_name,
    p.likes,
    COUNT(cp.id) AS comment_count,
    p.created_at,
    
    -- FACTOR 1: Engagement (1 point per like, 3 points per comment)
    (p.likes + (COUNT(cp.id) * 3)) AS engagement_score,

    -- FACTOR 2: Affinity (If the post author is a friend, give a massive 50 point bonus)
    IF(f.id IS NOT NULL, 50, 0) AS friend_bonus,

    -- FACTOR 3: Time Decay (Hours since the post was created)
    TIMESTAMPDIFF(HOUR, p.created_at, NOW()) AS hours_old,

    -- THE FINAL SCORE FORMULA
    (
        (p.likes + (COUNT(cp.id) * 3) + IF(f.id IS NOT NULL, 50, 0)) 
        / 
        POWER(TIMESTAMPDIFF(HOUR, p.created_at, NOW()) + 2, 1.5)
    ) AS feed_score

FROM posts p
JOIN profile prof ON p.posted_by = prof.id

-- Left join to count comments
LEFT JOIN post_comments cp ON p.id = cp.post_id

-- Left join to check if the author is friends with the current user (User ID 1)
LEFT JOIN friends f ON 
    (f.profile_id_1 = 1 AND f.profile_id_2 = p.posted_by) OR 
    (f.profile_id_1 = p.posted_by AND f.profile_id_2 = 1)

GROUP BY p.id, prof.name, p.created_at, f.id
ORDER BY feed_score DESC;
